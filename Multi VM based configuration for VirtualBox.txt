Vagrant.configure("2") do |config|
  # Define the 'web' machine
  config.vm.define "WEB" do |web|
    config.vm.box = "ubuntu/jammy64"
    config.vm.box_version = "20241002.0.0"
    config.vm.network :forwarded_port, guest: 22, host: 49152
    web.vm.hostname = "WEB"
    web.vm.network "private_network", ip: "x.x.x.x"
    
    web.vm.provider "virtualbox" do |v|
      v.memory = "2048"
      v.cpus = 2
    end
    
    web.vm.provision "shell", inline: <<-SHELL
      sudo apt-get update -y
      sudo apt install net-tools
      echo "Web server provisioned!"
    SHELL
  end

  # Define the 'db' machine
  config.vm.define "DB" do |db|
    config.vm.box = "ubuntu/jammy64"
    config.vm.box_version = "20241002.0.0"
    config.vm.network :forwarded_port, guest: 22, host: 49153
    db.vm.hostname = "DB"
    db.vm.network "private_network", ip: "x.x.x.x"
    
    db.vm.provider "virtualbox" do |v|
      v.memory = "2048"
      v.cpus = 2
    end
    
    db.vm.provision "shell", inline: <<-SHELL
      sudo apt-get update -y
      sudo apt install net-tools
      echo "Database server provisioned!"
    SHELL
  end
end
